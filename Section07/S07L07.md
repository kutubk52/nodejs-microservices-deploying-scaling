Create restock-mongodb.yml
```
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: restock-mongo-ebs-pvc
spec:
  accessModes:
    - ReadWriteOnce
  storageClassName: ebs-sc
  resources:
    requests:
      storage: 5Gi
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: restock-mongo-depl
spec:
  replicas: 1
  selector:
    matchLabels:
      app: restock-mongo
  template:
    metadata:
      labels:
        app: restock-mongo
    spec:
      containers:
        - name: restock-mongo
          image: mongo
          volumeMounts:
            - name: restock-mongo-storage
              mountPath: /data/db
          resources:
            requests:
              memory: "250Mi"
              cpu: "250m"
            limits:
              memory: "250Mi"
              cpu: "250m"
      volumes:
        - name: restock-mongo-storage
          persistentVolumeClaim:
            claimName: restock-mongo-ebs-pvc
---
apiVersion: v1
kind: Service
metadata:
  name: restock-mongo-srv
spec:
  selector:
    app: restock-mongo
  ports:
    - name: db
      protocol: TCP
      port: 27017
      targetPort: 27017
```

Apply the manifest:
```
kubectl apply -f restock-mongodb.yml
```

Create restock-depl.yml:
```
apiVersion: apps/v1
kind: Deployment
metadata:
  name: bevco-restock-depl
spec:
  replicas: 1
  selector:
    matchLabels:
      app: bevco-restock
  template:
    metadata:
      labels:
        app: bevco-restock
    spec:
      containers:
        - name: bevco-restock
          image: <image_repository>:latest
          env:
          - name: DATABASE
            value: 'mongodb://restock-mongo-srv:27017/restock'
          - name: RABBITMQ_URL
            value: 'rabbitmq-srv'
          - name: RABBITMQ_INVENTORY_QUEUE
            value: UpdateStock
          - name: RABBITMQ_USER
            valueFrom:
              secretKeyRef:
                name: rabbitmq-srv-default-user
                key: username
          - name: RABBITMC_PASSWORD
            valueFrom:
              secretKeyRef:
                name: rabbitmq-srv-default-user
                key: password
---
apiVersion: v1
kind: Service
metadata:
  name: bevco-restock-srv
spec:
  selector:
    app: bevco-restock
  ports:
    - name: bevco-restock
      protocol: TCP
      port: 3000
      targetPort: 3000
```

Apply the manifest:
```
kubectl apply -f restock-depl.yml
```
